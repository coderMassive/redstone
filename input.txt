// set r1 to how many times the loop should run
LDI r1 8 // find 8th fib number
RSH r1 r1 // divide by 2, as every loop is 2 numbers

// load first two fib numbers
LDI r2 1
LDI r3 1

LDI r4 1 // 4, count loading the first 2 fib numbers as a run

// exit loop if it ran r1 times
.loop CMP r4 r1
BRH = .exit

// calculate next 2 fib numbers
ADD r2 r3 r2
ADD r3 r2 r3

INC r4 // increment count
JMP .loop // loop back

.exit HLT
